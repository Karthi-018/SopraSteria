1. SELECT * FROM EMPLOYEES WHERE JOB='CLERK' AND YEAR(HIREDATE)>1997;

2. SELECT * FROM EMPLOYEES WHERE COMMISSION_PCT IS NOT NULL ORDER BY SALARY DESC;

3. SELECT ENAME, SALARY, SALARY*0.10 AS NEW_SALARY WHERE COMMISSION_PCT=0;

4. SELECT LAST_NAME,YEAR,MONTHS FROM EMPLOYEES ORDER BY YEAR DESC;

5. SELECT LAST_NAMR FROM EMPLOYEES WHERE LAST_NAME LIKE 'J%' OR LAST_NAME LIKE 'K%' OR LAST_NAME LIKE 'L%' OR LAST_NAME LIKE 'M%';

6. SELECT LAST_NAME, SALARY, DECODE( WHEN Commission>0 THEN 'YES' ELSE 'NO') AS COMMISSION FROM EMPLOYEES;

7. SELECT DEPARTMENT_NAME,LOCATION_ID,LAST_NAME,JOB_ID,SALARY FROM EMPLOYEES WHERE LOCATION_ID=&LOCATION_ID;

8. SELECT COUNT(EID) FROM EMPLOYEES WHERE LAST_NAME LIKE 'n';

9. SELECT DEPTID, DEPTNAME, LOCATION, COUNT(EID) FROM EMPLOYEES GROUP BY DEPTID;

10. SELECT JOB_ID FROM DEPARTMENT WHERE DEPT_ID IN (10,20);

11. SELECT JOB, Count(eid) FROM EMPLOYEES WHERE DEPARTMENT IN('Administrative','Executive') Group by department ORDER BY COUNT(EID);

12.SELECT LAST_NAME,HIRE_DATE FROM EMPLOYEES WHERE DAY(HIRE_DATE)<16;

13. SELECT LAST_NAME, SALARY, TRUNC(SALARY,-3) FROM EMPLOYEES;

14. SELECT E.ENAME,MANAGER_NAME,M.SALARY,GRADE_LEVEL FROM EMPLOYEE E JOIN MANAGER M USING(MANAGER_ID) WHERE M.SALARY>15000;

15. SELECT DEPT_ID,DEPT_NAME,COUNT(EID), AVG(SALARY),LAST_NAME FROM EMPLOYEES GROUP BY DEPARTMENT;
